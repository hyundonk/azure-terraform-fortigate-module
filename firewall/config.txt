config system probe-response
    set port 8008
    set mode http-probe
end

config system global
    set admin-sport 8443
    set admintimeout 30
    set timezone 60
end

config system settings
    set inspection-mode proxy
    set firewall-session-dirty check-new
    set default-voip-alg-mode kernel-helper-based
    set gui-load-balance enable
    set gui-allow-unnamed-policy enable
end

config system interface
    edit "port1"
        set allowaccess ping https ssh http fgfm probe-response
        set alias "Untrust"
    next
    edit "port2"
        set mode dhcp
        set allowaccess ping https ssh snmp http fgfm probe-response
        set alias "Trust"
        set defaultgw disable
        set dns-server-override disable
    next
end

config router static
    edit 1
        set dst 168.63.129.16 255.255.255.255
        set gateway 10.10.0.1
        set device "port1"
    next
    edit 2
        set dst 168.63.129.16 255.255.255.255
        set gateway 10.10.0.65
        set device "port2"
    next
    edit 3
        set dst 10.10.0.0 255.255.0.0
        set gateway 10.10.0.65
        set device "port2"
    next
end
